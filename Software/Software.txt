# Seleccionar el fragmento de la señal EMG
inicio_segundo = 5

t = np.arange(inicio_segundo, fin_segundo, 1/fs)  # Vector de tiempo de 0 a 1 segundo
f1 = 10  # Frecuencia de la señal en Hz
signal = fragmento_emg = Lectura_EMG[(Lectura_EMG.index >= inicio_segundo) & (Lectura_EMG.index < fin_segundo)]


# Calcular la FFT de la señal
fft_result = np.fft.fft(signal)
frequencies = np.fft.fftfreq(len(signal), d=1/fs)

# Calcular la magnitud en decibelios
magnitudes_db = -20*np.log10(np.abs(fft_result))

# Plotear la magnitud de la FFT en decibelios
plt.subplot(2, 1, 2)
plt.plot(frequencies[:len(frequencies)//2], magnitudes_db[:len(frequencies)//2])
